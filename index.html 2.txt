    <script>
        const startBtn = document.getElementById('start-btn');
        const welcomeScreen = document.getElementById('welcome-screen');
        const mainContent = document.getElementById('main-content');
        const dynamicText = document.getElementById('dynamic-text');
        const container = document.getElementById('canvas-container');
        const music = document.getElementById('bg-music');

        // Messages to cycle through (Translated to English)
        const messages = [
            "Today is a special day...",
            "Because it's your birthday!",
            "I hope you stay happy...",
            "And beautiful every single day.",
            "Always keep that smile on your face!",
            "Happy Birthday! ❤️"
        ];

        let messageIndex = 0;

        startBtn.addEventListener('click', () => {
            welcomeScreen.classList.add('hidden');
            mainContent.classList.remove('hidden');
            
            // Start music
            music.play().catch(e => console.log("Music autoplay blocked"));

            // Start falling text effect
            setInterval(createFallingText, 200);

            // Cycle through messages
            const messageInterval = setInterval(() => {
                if (messageIndex < messages.length) {
                    dynamicText.innerText = messages[messageIndex];
                    messageIndex++;
                } else {
                    clearInterval(messageInterval);
                }
            }, 3000);
        });

        function createFallingText() {
            const textElement = document.createElement('div');
            textElement.classList.add('falling-text');
            
            // Randomly choose between "Happy Birthday" or Hearts
            const items = ["Happy Birthday", "❤️", "Love", "✨", "Best Wishes"];
            textElement.innerText = items[Math.floor(Math.random() * items.length)];
            
            textElement.style.left = Math.random() * 100 + "vw";
            textElement.style.top = "-5vh";
            textElement.style.opacity = Math.random();
            textElement.style.fontSize = (Math.random() * 10 + 10) + "px";
            
            const duration = Math.random() * 3 + 2;
            textElement.style.animationDuration = duration + "s";
            
            container.appendChild(textElement);

            // Remove element after animation finishes
            setTimeout(() => {
                textElement.remove();
            }, duration * 1000);
        }
    </script>
